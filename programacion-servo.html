             void loop() {

               // Desplazamos a la posición 0º
                servoMotor.write(0);
                // Esperamos 1 segundo
                delay(1000);

                  // Esperamos 1 segundo

                    servoMotor.write(180);
                    // Esperamos 1 segundo
                    delay(1000);
